Für das folgende Tutorial wird immer dieser Initialisierungscode verwendet, an der Stelle //Code wird der Sourcecode aus den weiterführenden Tutorials eingefügt.

					//Initialisierung
var stage;
function init()
{
	stage = new createjs.Stage(document.getElementById("myCanvas"));
	createjs.Ticker.setFPS(30);
	createjs.Ticker.addEventListener("tick",updateStage,false);
	start();
}

function updateStage()
{
	stage.update();
}

{
	//Code
}

window.addEventListener("load",init,false);
